def get_average():
    with open('data.txt', 'r') as file:
        data = file.readlines()
# remove "temperatures heading to calc avg
    values = data[1:] #list slicing
    values = [float(i) for i in values] #list comprehension

    average_local = sum(values) / len(values)
    return average_local

average = get_average()
print(average)
